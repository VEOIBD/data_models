              } ),

                  tableClass: "table"

           targets: 0,
           render : function(data, type, row, meta){

                 return $('<a>')
                    .attr('href',row[7]+'/'+data)
                    .text(data)
                    .wrap('<div></div>')
                    .parent()
                    .html();}

                 return data;
              }
           }
        },
        {
          targets: [6,7],
            render : function(data, type, row, meta){

                 .text(data)
                 .wrap('<div></div>')
                 .parent()
                 .html();} 
           if(type === 'display' & data == 'Sage Bionetworks'){
               return $('<a>')
                  .attr('href', 'https://sagebionetworks.org/')
                  .text(data)
                  .wrap('<div></div>')
                  .parent()
                  .html();
           
           } else {
              return data;
           }
        }
     }
